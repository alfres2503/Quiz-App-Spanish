116 private void btnAgregar_Click(object sender, EventArgs e)
117 {
118
119 try
120 {
121 if (encabezadoFactura == null)
122 {
123 MessageBox.Show("Debe agregar los datos del encabezado de la factura para continuar", "Error", MessageBoxButtons.OK, MessageBoxIcon.Error);
124 return;
125 }
126
127 // Validar que el Producto ya no se haya agregado
128 if (encabezadoFactura._ListaFacturaDetalle.Count > 0)
129 {
130 foreach(var item in encabezadoFactura._ListaFacturaDetalle)
131 {
132 if (item.IDExamen == (cmbExamenes.SelectedItem as Examen).ID)
133 {
134 MessageBox.Show("El producto ya fue agregado previamente.", "Error", MessageBoxButtons.OK, MessageBoxIcon.Error);
135 return;
136 }
137 }
138 }
139
140 DetFactura detalleFactura = new DetFactura();
141
142 detalleFactura.IDFactura = encabezadoFactura.IDFactura;
143
144 detalleFactura.Secuencia = encabezadoFactura._ListaFacturaDetalle.Count == 0 ?
145 1 : encabezadoFactura._ListaFacturaDetalle.Max(p => p.Secuencia) + 1;
146
147 detalleFactura.IDExamen = (cmbExamenes.SelectedItem as Examen).ID;
148 detalleFactura.Monto = (cmbCertis.SelectedItem as Certificacion).Precio;
149 detalleFactura.ExamenRealizado = false;
150
151 encabezadoFactura.AgregarDetalle(detalleFactura);
152
153 string[] lineaFactura = {detalleFactura.Secuencia.ToString(),
154 detalleFactura.IDExamen.ToString(),
155 detalleFactura.Monto.ToString(),
156 (cmbExamenes.SelectedItem as Examen).CantPreguntas.ToString(),
157 (cmbExamenes.SelectedItem as Examen).Minutos.ToString() + " Minutos",
158 (cmbExamenes.SelectedItem as Examen).NotaMinima.ToString()};
159
160 dgvDetalle.Rows.Add(lineaFactura);
161 _MyLogControlEventos.Info($"Se añade: {lineaFactura}");
162 txtTotal.Text = encabezadoFactura.ObtenerTotal().ToString();
163 textBoxColones.Text = (encabezadoFactura.ObtenerTotal()*BancoC.valorDolar()).ToString();
164
165 this.cmbCertis.SelectedIndex = 0;
166 this.cmbExamenes.SelectedIndex = 0;
167
168 }
169 catch (Exception er)
170 {
171 MessageBox.Show(er.Message, "Error", MessageBoxButtons.OK, MessageBoxIcon.Error);
172 _MyLogControlEventos.Error("Ocurrió el siguiente error en facturación: " + er.Message);
173 }
